158    /**
159     * Procédure privée goRoom
160     * @param pCommand 
161     */
162    private void goRoom(final Command pCommand)
163    {
164        if ( ! pCommand.hasSecondWord() )
165        {
166            this.aGui.println( "Go where ?" );
167            return;
168        }
169        String vDirection = pCommand.getSecondWord();
170        
171        Room vNextRoom = this.aPlayer.getCurrentRoom().getExit(vDirection);
172
173        if (vNextRoom != null)
174        {
175            this.aPlayer.goRoomPlayer(vNextRoom);
176            this.printLocationInfo();
177        }
178        else
179            this.aGui.println( "There is no door !" );
180    }//goRoom(Command pDir)

189    /**
190     * Fonction booléenne privée processCommand : appeler la bonne méthode 
191     * en fonction de la commande passée en paramètre.
192     * @param pCommandLine
193     */
194    public void interpretCommand(final String pCommandLine)
195    {
196        this.aGui.println(">" + pCommandLine);
197        Command vCommand = this.aParser.getCommand( pCommandLine );
198        
199        if ( vCommand.isUnknown() )
200        {
201            this.aGui.println("I don't know what you mean ...");
202            return;
203        }
204        
205        String vCommandWord = vCommand.getCommandWord();
206        String vSecondWord = vCommand.getSecondWord();
207        
208        if( vCommandWord.equals( "help" ))
209        {
210            if(vSecondWord == null)
211            {
212                this.printHelp();
213            }
214            else if( vSecondWord.equals("go"))
215            {
216                this.aGui.println("go : this is a simple action to move.");
217            }
218            else
219            {
220                this.aGui.println("What is your problem ?");
221            }
222        }
223            
224            
225        else  if ( vCommandWord.equals( "go" ))
226        {
227            this.goRoom( vCommand );
228        }
229            
230            
231        else  if ( vCommandWord.equals( "look" ))
232        {
233            if(vSecondWord == null)
234            {
235                this.look();
236            }
237            else
238            {
239                if(this.aPlayer.getCurrentRoom().itemInHashMap(vSecondWord)) 
240                {
241                    this.aGui.println(this.aPlayer.getCurrentRoom().allInformationItem(vSecondWord));
242                }
243                else
244                {
245                    this.aGui.println("This item is not in the room !");
246                }
247            }
248        }
249        
250        else  if ( vCommandWord.equals( "eat" ))
251        {
252            if(vSecondWord == null)
253            {
254                this.eat();
255            }
256            else if (vSecondWord.equals("cookie"))
257            {
258                this.magicCookie();
259            }
260            else
261            {
262                this.aGui.println("What do you want ?");
263            }
264        }
265            
266        
267        else if ( vCommandWord.equals( "back" ))
268        {
269            if(vSecondWord == null)
270            {
271                this.back();
272            }
273            else
274            {
275                this.aGui.println("Just write  : back");
276            }
277        }
278        
279        else if (vCommandWord.equals( "test" ))
280        {
281            this.test(vSecondWord);
282        }
283        
284        else if (vCommandWord.equals( "drop"))
285        {
286            this.drop(vSecondWord);
287        }
288        else if (vCommandWord.equals( "take" ))
289        {
290            this.take(vSecondWord);
291        }
292        else if(vCommandWord.equals( "inventory"))
293        {
294            this.inventory();
295        }
296            
297            
298        else if ( vCommandWord.equals( "quit" ))
299        {
300            if ( vCommand.hasSecondWord() )
301                this.aGui.println("Quit what ?");
302            else
303                this.endGame();
304        }
305        
306        //this.timeLimit();
307    }//processCommand

338    /**
339     * Fonction test executant successivement toutes les commandes lues dans ce fichier de texte
340     * @param pNomFichier nom du fichier
341     */
342    public void test(final String pNomFichier)
343    {
344        if(pNomFichier == null)
345        {
346            this.aGui.println("Please enter the name of the file.");
347        }
348        else
349        {
350            Scanner vSc;
351            
352            java.io.InputStream vIS = this.getClass().getResourceAsStream( "/test/" + pNomFichier + ".txt" );             
353            if(vIS == null)
354            {
355                this.aGui.println("This file does not exist");
356            }
357            else
358            {
359                vSc = new Scanner(vIS);
360                while ( vSc.hasNextLine() ) 
361                {
362                    String vLigne = vSc.nextLine();
363                    this.interpretCommand(vLigne);
364                }
365            }
366        }
367    }//test

420    /**
421     * Procedure magicCookie qui double le poids maximal de l'inventaire du joueur. 
422     */
423    public void magicCookie()
424    {
425        Item vItem = this.aPlayer.getCurrentRoom().getItem("cookie");
426        if(vItem != null)
427        {
428            int weight = this.aPlayer.getMaxWeightInventory();
429            this.aPlayer.setMaxWeightInventory(2*weight);
430            this.aGui.println("You ate a magic cookie. \n Your maximum inventory weight has just increased.");
431            this.aPlayer.getCurrentRoom().removeItem("cookie");
432        }
433    }//magicCookie
