001/**
002 * Classe Item - un Item du jeu Welcome to the Castle
003 *
004 * @author DIDOUH Abel
005 * @date   04/01/2023
006 * @version Exercice 7.43
007 */
008public class Item
009{
010    /**
011     * Attributs (private)
012     */
013    private String aName;
014    private int aWeight;
015    private String aDescription;
016    /**
017     * Constructeur naturel de la classe Item
018     * @param pName     Name 
019     * @param pWeight   Weight
020     * @param pDescription Description
021     */
022    public Item(final String pName,final int pWeight, final String pDescription)
023    {
024        this.aName = pName;
025        this.aWeight = pWeight;
026        this.aDescription = pDescription;
027    }//Items()
028    /**
029     * Accesseur sur l'attribut aWeight
030     * @return int aWeight
031     */
032    public int getWeight()
033    {
034        return this.aWeight;
035    }//getWeight
036    /**
037     * Accesseur sur l'attribut aDescription
038     * @return String aDescription
039     */
040    public String getDescription()
041    {
042        return this.aDescription;
043    }//getDescription
044    /**
045     * Accesseur sur l'attribut aName
046     * @return String aName
047     */
048    public String getName()
049    {
050        return this.aName;
051    }//getName
052    /**
053     * Fonction qui renvoie la description (Name - Weight - Description) d'un item
054     * @return String Name Weight Description
055     */
056    public String getLongDescription()
057    {
058        return  "Name : " + this.getName() + "\n" +
059                "Weight : " + this.getWeight() + "\n" +
060                "Description : " + this.getDescription();
061    }//getLongDescription
062}
