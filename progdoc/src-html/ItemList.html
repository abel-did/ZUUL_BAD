001import java.util.HashMap;
002import java.util.Set;
003/**
004 * Classe ItemList - Les items du jeu
005 *
006 * @author DIDOUH Abel
007 * @date   04/01/2023
008 * @version Exercice 7.43
009 */
010public class ItemList
011{
012    /**
013     * Attributs (private)
014     */
015    private HashMap<String, Item> aItems;
016    /**
017     * Constructeur par d√©faut
018     */
019    public ItemList()
020    {
021        this.aItems = new HashMap<String, Item>();
022    }
023    /**
024     * Accesseur sur l'attribut aItems
025     * @param pName nom de l'objet
026     * @return Item aItems
027     */
028    public Item getItem(final String pName)
029    {
030        return this.aItems.get(pName);
031    }
032    /**
033     * Procedure qui ajoute un item
034     * @param pName nom de l'objet
035     * @param pItem objet
036     */
037    public void putItem(final String pName, final Item pItem)
038    {
039        this.aItems.put(pName, pItem);
040    }
041    /**
042     * Procedure qui supprime un item
043     * @param pName nom de l'objet
044     */
045    public void removeItem(final String pName)
046    {
047        this.aItems.remove(pName);
048    }
049    /**
050     * Fonction qui renvoie une chaine comprennant les items
051     */
052    public String getItemsString()
053    {
054        if(this.aItems.isEmpty())
055        {
056            return "No item here.";
057        }
058        else
059        {
060            String vReturnString = "Items : ";
061            Set<String> keys = aItems.keySet();
062            for(String aItems : keys)
063            {
064                vReturnString += " " + aItems;
065            }
066            return vReturnString;
067        }
068    }
069    /**
070     * Procedure qui verifie si l'item est dans la liste
071     * @param pItemName nom de l'objet
072     */
073    public boolean itemInList(final String pItemName)
074    {
075        if(this.aItems.containsKey(pItemName))
076        {
077            return true;
078        }
079        else
080        {
081            return false;
082        }
083    }
084    /**
085     * Fonction qui retourne l'inventaire du joueur.
086     */
087    final String inventory()
088    {
089        String vReturnString = "Items : \n";
090        Set<String> keys = aItems.keySet();
091        
092        if(!this.aItems.isEmpty())
093        {
094            for(String aItems : keys)
095            {
096                vReturnString += "\t" + aItems + " \n"; 
097            }
098            return vReturnString;
099        }
100        else
101        {
102            return ("No items in the inventory.\n");
103        }
104    }//Inventory
105}
