001/**
002 * Classe Command - une commande du jeu d'aventure Zuul.
003 *
004 * @author DIDOUH Abel
005 * @date   04/01/2023
006 * @version Exercice 7.43
007 */
008public class Command
009{
010    /**
011     * Attributs (private)
012     */
013    private String aCommandWord;
014    private String aSecondWord;
015    
016    /**
017     * Constructeur naturel
018     * @param pCommandWord chaine pour initialiser aCommandWord
019     * @param pSecondWord chaine pour initialiser aSecondWord
020     */
021    public Command(final String pCommandWord, final String pSecondWord)
022    {
023        this.aCommandWord = pCommandWord;
024        this.aSecondWord  = pSecondWord; 
025    }//Command()
026    /**
027     * Accesseur aCommandWord
028     * @return String aCommandWord
029     */
030    public String getCommandWord()
031    {
032        return this.aCommandWord;
033    }//getCommandWord
034    /**
035     * Accesseur aSecondWord
036     * @return String aSecondWord
037     */
038    public String getSecondWord()
039    {
040        return this.aSecondWord;
041    }//getSecondWord
042    /**
043     * Fonction booléenne qui permettra de vérifier qu'un second 
044     * mot a bien été tapé.
045     * @return boolean
046     */
047    public boolean hasSecondWord()
048    {
049        return this.aSecondWord != null;
050    }//hasSecondWord
051    /**
052     * Fonction booléenne qui retourne vrai si le premier mot est null.
053     * @return boolean
054     */
055    public boolean isUnknown()
056    {
057        return this.aCommandWord == null;
058    }//isUnknown
059} // Command
